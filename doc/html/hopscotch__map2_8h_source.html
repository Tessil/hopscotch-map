863  template<typename U = value_type,
864  typename std::enable_if<std::is_nothrow_move_constructible<U>::value>::type* = nullptr>
865  void rehash_internal(size_type count) {
866  /*
867  * Little more complicate here. We want to avoid any exception so we can't just insert elements
868  * in the map trough insert_internal. The method may throw even if value_type is nothrow_move_constructible
869  * due to the m_overflow_elements.push_back in the method.
870  *
871  * So we first move safely all the elements from m_buckets in the new map (we know they will not go into
872  * overflow). We then swap m_overflow_elements into the new map and we update all necessary overflow flags.
873  *
874  * This way we don't do any memory allocation (outside the construction of m_buckets) and we avoid
875  * any exception which may hinder the strong exception-safe guarantee.
876  */
877  hopscotch_hash tmp_map(count, m_hash, m_key_equal, get_allocator(), m_max_load_factor);
878 
879  for(hopscotch_bucket& bucket : m_buckets) {
880  if(bucket.is_empty()) {
881  continue;
882  }
883 
884  const std::size_t ibucket_for_hash = tmp_map.bucket_for_hash(
885  tmp_map.m_hash(KeySelect()(bucket.get_value())));
886  tmp_map.insert_internal(std::move(bucket.get_value()), ibucket_for_hash);
887  }
888 
889  tsl_assert(tmp_map.m_overflow_elements.empty());
890  if(!m_overflow_elements.empty()) {
891  tmp_map.m_overflow_elements.swap(m_overflow_elements);
892  tmp_map.m_nb_elements += tmp_map.m_overflow_elements.size();
893 
894  for(const value_type& value : tmp_map.m_overflow_elements) {
895  const std::size_t ibucket_for_hash = tmp_map.bucket_for_hash(tmp_map.m_hash(KeySelect()(value)));
896  tmp_map.m_buckets[ibucket_for_hash].set_overflow(true);
897  }
898  }
899 
900  tmp_map.swap(*this);
901  }
902 
903  /*
904  * Find in m_overflow_elements an element for which the bucket it initially belongs to,
905  * equals original_bucket_for_hash.
906  *
907  * Return m_overflow_elements.end() if none.
908  */
909  iterator_overflow find_in_overflow_from_bucket(iterator_overflow search_start,
910  std::size_t original_bucket_for_hash)
911  {
912  for(auto it = search_start; it != m_overflow_elements.end(); ++it) {
913  const std::size_t bucket_for_overflow_hash = bucket_for_hash(m_hash(KeySelect()(*it)));
914  if(bucket_for_overflow_hash == original_bucket_for_hash) {
915  return it;
916  }
917  }
918 
919  return m_overflow_elements.end();
920  }

930  m_buckets[ibucket_for_hash].set_overflow(false);

977  template<typename P>
978  std::pair<iterator, bool> insert_internal(P&& value, std::size_t ibucket_for_hash) {
979  if((m_nb_elements - m_overflow_elements.size() + 1) > m_load_threshold) {
980  rehash_internal(get_expand_size());
981  ibucket_for_hash = bucket_for_hash(m_hash(KeySelect()(value)));
982  }
983 
984  std::size_t ibucket_empty = find_empty_bucket(ibucket_for_hash);
985  if(ibucket_empty < m_buckets.size()) {
986  do {
987  // Empty bucket is in range of NeighborhoodSize, use it
988  if(ibucket_empty - ibucket_for_hash < NeighborhoodSize) {
989  auto it = insert_in_bucket(std::forward<P>(value), ibucket_empty, ibucket_for_hash);
990  return std::make_pair(iterator(it, m_buckets.end(), m_overflow_elements.begin()), true);
991  }
992  }
993  // else, try to swap values to get a closer empty bucket
994  while(swap_empty_bucket_closer(ibucket_empty));
995  }
996 
997  // A rehash will not change the neighborhood, put the value in overflow list
998  if(!will_neighborhood_change_on_rehash(ibucket_for_hash)) {
999  m_overflow_elements.push_back(std::forward<P>(value));
1000  m_buckets[ibucket_for_hash].set_overflow(true);
1001  m_nb_elements++;
1002 
1003  return std::make_pair(iterator(m_buckets.end(), m_buckets.end(),
1004  std::prev(m_overflow_elements.end())), true);
1005  }
1006 
1007  rehash_internal(get_expand_size());
1008 
1009  ibucket_for_hash = bucket_for_hash(m_hash(KeySelect()(value)));
1010  return insert_internal(std::forward<P>(value), ibucket_for_hash);
1011  }

1050  *
1051  * Return bucket iterator to ibucket_empty
1052  */
1053  template<typename P>
1054  iterator_buckets insert_in_bucket(P&& value, std::size_t ibucket_empty, std::size_t ibucket_for_hash) {
1055  tsl_assert(ibucket_empty >= ibucket_for_hash );
1056  tsl_assert(m_buckets[ibucket_empty].is_empty());
1057  m_buckets[ibucket_empty].set_value(std::forward<P>(value));
1058 
1059  tsl_assert(!m_buckets[ibucket_for_hash].is_empty());
1060  m_buckets[ibucket_for_hash].toggle_neighbor_presence(ibucket_empty - ibucket_for_hash);

1062

1066  /*
1067  * Try to swap the bucket ibucket_empty_in_out with a bucket preceding it while keeping the neighborhood
1068  * conditions correct.
1069  *
1070  * If a swap was possible, the position of ibucket_empty_in_out will be closer to 0 and true will re returned.
1071  */
1072  bool swap_empty_bucket_closer(std::size_t& ibucket_empty_in_out) {
1073  tsl_assert(ibucket_empty_in_out >= NeighborhoodSize);
1074  const std::size_t neighborhood_start = ibucket_empty_in_out - NeighborhoodSize + 1;
1075 
1076  for(std::size_t to_check = neighborhood_start; to_check < ibucket_empty_in_out; to_check++) {
1077  neighborhood_bitmap neighborhood_infos = m_buckets[to_check].get_neighborhood_infos();
1078  std::size_t to_swap = to_check;
1079 
1080  while(neighborhood_infos != 0 && to_swap < ibucket_empty_in_out) {
1081  if((neighborhood_infos & 1) == 1) {
1082  tsl_assert(m_buckets[ibucket_empty_in_out].is_empty());
1083  tsl_assert(!m_buckets[to_swap].is_empty());
1084 
1085  m_buckets[to_swap].swap_value_into_empty_bucket(m_buckets[ibucket_empty_in_out]);
1086 
1087  tsl_assert(!m_buckets[to_check].check_neighbor_presence(ibucket_empty_in_out - to_check));
1088  tsl_assert(m_buckets[to_check].check_neighbor_presence(to_swap - to_check));
1089 
1090  m_buckets[to_check].toggle_neighbor_presence(ibucket_empty_in_out - to_check);
1091  m_buckets[to_check].toggle_neighbor_presence(to_swap - to_check);
1092 
1093 
1094  ibucket_empty_in_out = to_swap;
1095 
1096  return true;
1097  }
1098 
1099  to_swap++;
1100  neighborhood_infos = static_cast<neighborhood_bitmap>(neighborhood_infos >> 1);
1101  }
1102  }
1103 
1104  return false;
1105  }
